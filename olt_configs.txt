# ==========================================КОД РЕЄСТРАЦІЇ ОНУ НА ОЛТУ==========================================
[GPON ONU REG]
conf t
interface gpon-olt_{PORT_BASE} 
onu {ONU_ID} type GPON-1GE sn {SN}
exit
interface gpon-onu_{PORT} 

[EPON ONU REG]
conf t
interface epon-olt_{PORT_BASE}
onu {ONU_ID} type NTS-101 mac {MAC}
exit
interface epon-onu_{PORT}
# ==========================================КОД ЗАМІНИ ОНУ==========================================
[GPON ONU SWITCH]
conf t
interface gpon-olt_{PORT_BASE} 
no onu {ONU_ID}
onu {ONU_ID} type GPON-1GE sn {SN}
exit
interface gpon-onu_{PORT} 

[EPON ONU SWITCH]
conf t
interface epon-olt_{PORT_BASE}
no onu {ONU_ID}
onu {ONU_ID} type NTS-101 mac {MAC}
exit
interface epon-onu_{PORT}
# ==========================================КОД PON ONU==========================================
[GPON PON-ONU]
exit
pon-onu-mng gpon-onu_{PORT}
loop-detect ethuni eth_0/1 enable
gemport 1 flow 1
vlan gemport 1 mode transparent
end

[EPON PON-ONU]
exit
pon-onu-mng epon-onu_{PORT}
auto-config
compatibility enable mode CTC
vlan port eth_0/1 mode transparent
interface-loopdetect eth_0/1 activate
end
