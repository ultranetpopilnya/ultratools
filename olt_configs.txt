[TEMPLATE: ZTE C300 GPON]
  description {LOGIN}
  tcont 1 name gpon-home profile {SPEED}
  gemport 1 tcont 1
  gemport 1 traffic-limit upstream {SPEED} downstream {SPEED} 
  service-port 1 vport 1 user-vlan untag vlan {VLAN} 
  ip dhcp snooping quota 1 vport 1
  ip dhcp snooping enable vport 1
  port-identification format DSL-FORUM-PON vport 1
  port-identification sub-option remote-id enable vport 1
  port-identification sub-option remote-id name {LOGIN} vport 1
  dhcpv4-l2-relay-agent enable vport 1
  dhcpv4-l2-relay-agent trust true replace vport 1
exit
pon-onu-mng gpon-onu_{PORT}
loop-detect ethuni eth_0/1 enable
gemport 1 flow 1
vlan gemport 1 mode transparent
end
